var addbaseinfo = function(moduleName, methodName, body) {
    var url = eo.env.http.baseUrl.get() + "/FHH/EM1HESERVICE2/eservice/" + moduleName + "/" + methodName + "?_fs_token=" + fs_token;
    var api_addbaseinfo = {
        "url": url,
        "method": "POST",
        "headers": {
            "Content-Type": "application/json",
            "Cookie": cookie
        },
        "body": body,
        "timelimit": 20000
    };
    var result_addbaseinfo = eo.execute(api_addbaseinfo);
    var response_addbaseinfo = JSON.parse(result_addbaseinfo.response);

    if (response_addbaseinfo !== null) {
        if (response_addbaseinfo.Result['FailureCode'] === 0) {
            eo.info("数据创建成功！");
            return response_addbaseinfo;

        } else {
            eo.info("数据创建失败，失败原因：" + response_addbaseinfo.Result['FailureMessage']);
            return response_addbaseinfo;
        }
    } else {
        eo.info("新建数据失败！");
        eo.stop("停止回调信息");
    }

};
